exports.start = function (pathParams) {
    // rootPath为项目根目录
    rootPath = pathParams
    const beginPath = path.join(rootPath, 'node_modules')
    if (!fs.existsSync(beginPath)) {
        console.log(`路径${beginPath}不存在，执行结束`.red)
        return;
    }
    const consoleLogPath = path.join(rootPath, 'log');
    fs.appendFileSync(consoleLogPath, `以下为新发布成功的包\n`);
    // 循环publish
    loop(beginPath)
    console.log('done'.green)
}

function loop(currentPath) {
    // 读取所有子模块path
    const childModulesPath = fs.readdirSync(currentPath)
    // 子模块path循环
    childModulesPath.forEach(childModulePath => {
        // 解析子模块路径信息
        let childModulePathInfo = null
        let currentChildModulePath = path.join(currentPath, childModulePath)
        childModulePathInfo = fs.statSync(currentChildModulePath);
        // 如果是目录且不是.bin这种生成的文件，则为模块，否则可能是其他类型文件
        if (childModulePathInfo.isDirectory() && childModulePath.startsWith('.') !== true) {
            if (childModulePath.startsWith('@')) {
                // 是私域目录则进入循环其下子模块，等效于node_modules下的结构，所以此时可以递归执行
                loop(currentChildModulePath)
            } else {
                // 是普通模块
                // 1、推送当前包
                runPublish(currentChildModulePath)
                // 2、其下是否有node_modules,有则进行递归
                const littleChildModulesPath = fs.readdirSync(currentChildModulePath)
                if (littleChildModulesPath.includes('node_modules')) {
                    let littleChildModulePath = path.join(currentChildModulePath, 'node_modules')
                    loop(littleChildModulePath)
                }
            }
        }
    })
}

function runPublish(modulePath) {
    const consoleLogPath = path.join(rootPath, 'log');
    // 如果是目录，则执行下述操作
    // 当前包目录下的package.json文件路径
    const packageJsonPath = path.join(modulePath, 'package.json');
    if (fs.existsSync(packageJsonPath)) {
        // 读取当前包目录下的package.json文件的内容，返回字符串
        const packageJsonContentString = fs.readFileSync(packageJsonPath, 'utf8');
        // 解析package.json文件的内容成json格式的对象
        const parsedPackJson = JSON.parse(packageJsonContentString);
        // 将package.json文件内的scripts的内容置为空，防止因为出现的prepublish等字段打断publish进程
        if (parsedPackJson.scripts && Object.keys(parsedPackJson.scripts).length !== 0) {
            parsedPackJson.scripts = {};
        }
        if (parsedPackJson.publishConfig && Object.keys(parsedPackJson.publishConfig).length !== 0) { 
            parsedPackJson.publishConfig = {}
        }
        fs.writeFileSync(packageJsonPath, JSON.stringify(parsedPackJson));
        // 把当前的目录切换到当前包路径下
        process.chdir(modulePath);
        // 在包路径下执行cmd控制台命令 npm publish
        try {
            execSync('npm publish', { stdio: 'ignore' });
            console.log(`${parsedPackJson.name}:${parsedPackJson.version}`, ' 【 发布成功 】'.green);
            fs.appendFileSync(consoleLogPath, `${parsedPackJson.name}@${parsedPackJson.version}\n`);
        } catch {
            console.log(`${parsedPackJson.name}:${parsedPackJson.version}`, ' 【 已存在 】'.yellow);
        }
        process.chdir(rootPath);
        // 如果解析好的json格式的对象存在scripts属性且该属性不为空对象，
        // 那么将其内容设置成空对象,防止因为存在pre类的命令导致触发其他的脚本执行
    }
}
